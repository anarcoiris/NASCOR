SELECT MAX(price) AS precio_maximo
FROM Airbnb

#precio_maximo: 9000

SELECT COUNT(*) AS apartamentos_baratos
FROM airbnb
WHERE price < 150;

#apartamentos_baratos: 21510

SELECT neighbourhood, AVG(price) AS precio_medio
FROM airbnb
GROUP BY neighbourhood
ORDER BY precio_medio DESC
LIMIT 3;
